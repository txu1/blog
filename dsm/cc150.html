// O(n) O(n)
function isUnique(astr) {
    return astr ? new Set(astr).size === astr.length : true
}

// O(n) O(n)
function isUnique(astr) {
    if (!astr) return true
    let map = new Map()
    for (let char of astr) {
        if (!map.get(char)) map.set(char, 1)
        else return false
    }
    return true
}

// O(n^2) O(1)
function isUnique(astr) {
    if (!astr) return true
    for (let i = 0; i < astr.length; i++) {
        if (i != lastIndexOf(astr, astr[i])) return false
    }
    return true
}

function lastIndexOf(astr, target) {
    let i = astr.length
    while (i--) {
        if (astr[i] === target) return i
    }
    return -1
}

# 判定是否互为字符重排


// O(n) O(n)
function CheckPermutation(s1, s2) {
    if (s1.length !== s2.length) return false
    let map = new Map()
    for (char of s1) {
        !map.has(char) && map.set(char, 1)
        map.set(char, map.get(char) + 1)
    }
    for (char of s2) {
        if (map.get(char)) map.set(char, map.get(char) - 1)
        else return false
    }
    return true
}

// O(n^2) O(1)
function CheckPermutation(s1, s2) {
    if (s1.length !== s2.length) return false
    while (s1.length) {
        let ele = s1[0], index = s2.indexOf(ele)
        if (index != -1) {
            s1 = s1.slice(1)
            s2 = s2.slice(0, index) + s2.slice(index + 1)
        } else {
            return false
        }
    }
    return true
}

// O(nlogn) O(n)
function CheckPermutation(s1, s2) {
    if (s1.length !== s2.length) return false
    return s1.split('').sort().join('') == s2.split('').sort().join('')
}

# 一次编辑


// O(n) O(1)
function oneEditAway(first, second) {
    if (first === second) return true
    let len1 = first.length, len2 = second.length
    if (len1 === len2) {
        let diff = 0;
        for (let i = 0; i < len1; i++) {
            if (first[i] !== second[i]) diff++
        }
        return diff <= 1
    } else if (len1 - len2 === 1 || len1 - len2 === -1) {
        let s1 = len1 > len2 ? first : second, s2 = len1 > len2 ? second : first
        for (let i = 0; i < s1.length; i++) {
            if (s1[i] !== s2[i]) return s1.substring(i + 1) === s2.substring(i)
        }
    }
    return false
}

# 字符串压缩


// O(n) O(n)
function compressString(S) {
    let res = '', sReg = S.match(/([A-z])\1*/g)
    if (sReg) {
        sReg.forEach(e => {
            res += e.charAt(0) + e.length
        })
    }
    return res.length < S.length ? res : S
}

// O(n) O(n)
function compressString(S) {
    let count = 1, res = ''
    for (let i = 1; i < S.length + 1; i++) {
        if (S[i - 1] === S[i]) {
            count++
        } else {
            res += S.slice(i - 1, i) + count
            count = 1
        }
    }
    return S.length > res.length ? res : S
}

// O(n^2) O(1)
function rotate(matrix) {
    let len = matrix.length
    for (let i = 0; i < len; i++) {
        for (let j = i; j < len; j++) {
            [matrix[i][j], matrix[j][i]] = [matrix[j][i], matrix[i][j]]
        }
    }
    matrix.forEach(row => row.reverse())
}

# 零矩阵


// O(m*n) O(m+n)
function setZeroes(matrix) {
    let h = matrix.length, w = matrix[0].length
    let row = Array(h), col = Array(w)
    for (let i = 0; i < h; i++) {
        for (let j = 0; j < w; j++) {
            if (matrix[i][j] == 0) {
                row[i] = true
                col[j] = true
            }
        }
    }
    for (let i = 0; i < h; i++) {
        for (let j = 0; j < w; j++) {
            if (row[i] || col[j]) {
                matrix[i][j] = 0
            }
        }
    }
}

function isFlipedString(s1, s2) {
    if (s1 === s2) return true
    if (s1.length != s2.length) return false
    s1 += s1
    let p1 = p2 = 0
    while (p1 != s1.length && p2 != s2.length) {
        if (s1[p1] != s2[p2]) {
            p1++
            p2 = 0
        } else {
            p1++
            p2++
        }
    }
    return p2 === s2.length
}

// O(n) O(1)
function isPalindrome(head) {
    if (!head) return true
    let mid = findMidNode(head)
    let cur1 = head, cur2 = reverseLink(mid)
    while (cur2) {
        if (cur1.val != cur2.val) return false
        cur1 = cur1.next
        cur2 = cur2.next
    }
    return true
}

function findMidNode(head) {
    let slow = fast = head
    while (fast && fast.next) {
        fast = fast.next.next
        slow = slow.next
    }
    return slow
}

function reverseLink(head) {
    let prev = null, cur = head
    while (cur) {
        let nextTemp = cur.next
        cur.next = prev
        prev = cur
        cur = nextTemp
    }
    return prev
}

// O(n) O(n)
function getIntersectionNode(headA, headB) {
    if (!headA || !headB) return null
    let set = new Set()
    while (headA) {
        if (set.has(headA)) headA = headA.next
        else set.add(headA)
    }
    while (headB) {
        if (set.has(headB)) return headB
        else headB = headB.next
    }
    return null
}

// O(n^2) O(1)
function getIntersectionNode(headA, headB) {
    if (!headA || !headB) return null
    let cur1 = headA
    while (cur1) {
        let cur2 = headB
        while (cur2) {
            if (cur1 == cur2) return cur1
            cur2 = cur2.next
        }
        cur1 = cur1.next
    }
    return null
}

// O(n) O(1)
function getIntersectionNode(headA, headB) {
    if (!headA || !headB) return null
    let cur1 = headA, cur2 = headB
    while (cur1 != cur2) {
        cur1 = cur1 == null ? headB : cur1.next
        cur2 = cur2 == null ? headA : cur2.next
    }
    return cur1
}

# 环路检测


// O(n) O(n)
function detectCycle(head) {
    if (!head) return null
    let set = new Set(), cur = head
    while (cur) {
        if (!set.has(cur)) set.add(cur)
        else return cur
        cur = cur.next
    }
    return cur
}

// O(n) O(1)
function detectCycle(head) {
    if (!head) return null
    let fast = slow = head
    while (fast) {
        if (!fast.next || !fast.next.next) return null
        fast = fast.next.next
        slow = slow.next
        if (fast == slow) break
    }
    slow = head
    while (slow != fast) {
        slow = slow.next
        fast = fast.next
    }
    return fast
}

# 三合一


class TripleInOne {
    constructor(stackSize) {
        this.arr = Array(3 + 3 * stackSize).fill(0)
        this.size = stackSize
    }
    push(stackNum, value) {
        if (this.arr[stackNum] == this.size) return
        this.arr[stackNum * this.size + 3 + this.arr[stackNum]++] = value
    }
    pop(stackNum) {
        if (this.arr[stackNum] == 0) return -1
        return this.arr[stackNum * this.size + 3 + --this.arr[stackNum]]
    }
    peek(stackNum) {
        if (this.arr[stackNum] == 0) return -1
        return this.arr[stackNum * this.size + 3 + this.arr[stackNum] - 1]
    }
    isEmpty(stackNum) {
        return this.arr[stackNum] == 0
    }
}

# 栈的最小值


class MinStack {
    constructor() {
        this.stack = []
        this.minStack = []
    }
    push(x) {
        this.stack.push(x)
        let msLen = this.minStack.length
        if (!msLen || x <= this.minStack[msLen - 1]) this.minStack.push(x)
    }
    pop() {
        let res = this.stack.pop()
        return res == this.minStack.slice(-1)[0] ? this.minStack.pop() : res
    }
    top() {
        return this.stack.slice(-1)[0]
    }
    getMin() {
        return this.minStack.slice(-1)[0]
    }
}

class MinStack {
    constructor() {
        this.stack = []
        this.min = null
    }
    push(x) {
        if (this.min == null || x <= this.min) {
            if (this.stack.length) this.stack.push(this.min)
            this.min = x
        }
        this.stack.push(x)
    }
    pop() {
        if (this.stack.length == 1) this.min = null
        else if (this.top() == this.min) {
            this.stack.pop()
            this.min = this.top()
        }
        return this.stack.pop()
    }
    top() {
        return this.stack.slice(-1)[0]
    }
    getMin() {
        return this.min
    }
}

# 堆盘子


class StackOfPlates {
    constructor(cap) {
        this.cap = cap
        this.arr = []
    }
    push(val) {
        if (this.cap <= 0) return
        if (!this.arr.length || this.arr.slice(-1)[0].length == this.cap) this.arr.push([])
        this.arr.slice(-1)[0].push(val)
    }
    pop() {
        return this.popAt(this.arr.length - 1)
    }
    popAt(index) {
        if (!this.arr.length || index >= this.arr.length) return -1
        let res = this.arr[index].pop()
        if (!this.arr[index].length) this.arr.splice(index, 1)
        return res
    }
}

# 栈排序


class SortedStack {
    constructor() {
        this.stack1 = []
        this.stack2 = []
    }
    push(val) {
        if (this.isEmpty()) this.stack1.push(val)
        else {
            while (!this.isEmpty() && this.peek() < val) {
                this.stack2.push(this.pop())
            }
            this.stack1.push(val)
            while (this.stack2.length) {
                this.stack1.push(this.stack2.pop())
            }
        }
    }
    pop() {
        return this.isEmpty() ? null : this.stack1.pop()
    }
    peek() {
        return this.isEmpty() ? -1 : this.stack1.slice(-1)[0]
    }
    isEmpty() {
        return this.stack1.length == 0
    }
}

class SortedStack {
    constructor() {
        this.stack1 = []
        this.stack2 = []
    }
    push(val) {
        while (true) {
            let max = !this.stack1.length ? Infinity : this.stack1.slice(-1)[0]
            let min = !this.stack2.length ? -Infinity : this.stack2.slice(-1)[0]
            if (val > max) {
                this.stack2.push(this.stack1.pop())
            } else if (val < min) {
                this.stack1.push(this.stack2.pop())
            } else {
                this.stack1.push(val)
                break
            }
        }
    }
    pop() {
        while (this.stack2.length) this.stack1.push(this.stack2.pop())
        return this.stack1.length ? this.stack1.pop() : null
    }
    peek() {
        while (this.stack2.length) this.stack1.push(this.stack2.pop())
        return this.stack1.length ? this.stack1.slice(-1)[0] : -1
    }
    isEmpty() {
        return this.stack1.length == 0 && this.stack2.length == 0
    }
}

# 动物收容所


class AnimalShelf {
    constructor() {
        this.dogs = []
        this.cats = []
    }
    enqueue(animal) {
        animal[1] ? this.dogs.push(animal) : this.cats.push(animal)
    }
    dequeueAny() {
        if (!this.dogs.length && !this.cats.length) return [-1, -1]
        else if (!this.cats.length) return this.dogs.shift()
        else if (!this.dogs.length) return this.cats.shift()
        else return this.dogs[0][0] > this.cats[0][0] ? this.cats.shift() : this.dogs.shift()
    }
    dequeueDog() {
        return this.dogs.length ? this.dogs.shift() : [-1, -1]
    }
    dequeueCat() {
        return this.cats.length ? this.cats.shift() : [-1, -1]
    }
}

// 广度优先版
function findWhetherExistsPath(n, graph, start, target) {
    let map = new Map()
    for (let edge of graph) {
        if (map.has(edge[0])) map.get(edge[0]).push(edge[1])
        else map.set(edge[0], [edge[1]])
    }
    return bfs(map, start, target)
}

function bfs(map, start, target) {
    if (map.get(start) != null) {
        for (let i of map.get(start)) {
            if (i == target) return true
        }
        for (let j of map.get(start)) {
            if (bfs(map, j, target)) return true
        }
    }
    return false
}

// 深度优先版
function findWhetherExistsPath(n, graph, start, target) {
    let map = new Map()
    for (let edge of graph) {
        if (map.has(edge[0])) map.get(edge[0]).push(edge[1])
        else map.set(edge[0], [edge[1]])
    }
    return dfs(map, start, target)
}

function dfs(map, start, target) {
    if (map.get(start) != null) {
        for (let i of map.get(start)) {
            if (i == target) return true
            if (dfs(map, i, target)) return true
        }
    }
    return false
}

// 集合版
function findWhetherExistsPath(n, graph, start, target) {
    let set = new Set(), count = 0
    set.add(start)
    for (let arr of graph) {
        if (set.has(arr[0])) set.add(arr[1])
    }
    return set.has(target)
}

# 最小高度树


// O(n) O(n)
function sortedArrayToBST(nums) {
    function dfs(left, right) {
        if (left >= right) return null
        let mid = Math.floor((left + right) / 2)
        let root = new TreeNode(nums[mid])
        root.left = dfs(left, mid)
        root.right = dfs(mid + 1, right)
        return root
    }
    return dfs(0, nums.length)
}

# 特定深度节点链表


// O(n) O(n)
function listOfDepth(tree) {
    let res = []
    function dfs(node, level) {
        if (!node) return null
        if (res.length == level) res.push(new ListNode(node.val))
        else {
            let head = new ListNode(node.val)
            head.next = res[level]
            res[level] = head
        }
        dfs(node.right, level + 1)
        dfs(node.left, level + 1)
    }
    dfs(tree, 0)
    return res
}

# 检查平衡性


// O(n) O(n)
function isBalanced(root) {
    let res = true
    function dfs(node) {
        if (node == null) return 0
        const left = dfs(node.left)
        const right = dfs(node.right)
        Math.abs(left - right) > 1 && (res = false)
        return Math.max(left, right) + 1
    }
    dfs(root)
    return res
}

# 合法二叉搜索树


// O(n) O(n)
function isValidBST(root) {
    let nodes = [], pre = -Infinity
    function dfs(node) {
        if (node) {
            dfs(node.left)
            nodes.push(node.val)
            dfs(node.right)
        }
    }
    dfs(root)
    for (let i = 0; i < nodes.length; i++) {
        if (nodes[i] > pre) pre = nodes[i]
        else return false
    }
    return true
}

// O(n) O(n)
function isValidBST(root) {
    if (!root) return true
    let prev = -Infinity
    function dfs(node) {
        if (!node) return true
        if (dfs(node.left)) {
            if (node.val > prev) {
                prev = node.val
                return dfs(node.right)
            } else return false
        } else return false
    }
    return dfs(root)
}

# 后继者


// 递归版
function inorderSuccessor(root, p) {
    if (!root || p == null) return null
    if (p.val >= root.val) return inorderSuccessor(root.right, p)
    else return inorderSuccessor(root.left, p) ? inorderSuccessor(root.left, p) : root
}

// 迭代版
function inorderSuccessor(root, p) {
    if (p.right) {
        p = p.right
        while (p.left) p = p.left
        return p
    }
    let res = null
    while (root != p) {
        if (root.val < p.val) root = root.right
        else {
            res = root
            root = root.left
        }
    }
    return res
}

# 首个共同祖先


// O(n) O(n)
function lowestCommonAncestor(root, p, q) {
    if (!root) return null
    let left = lowestCommonAncestor(root.left, p, q)
    let right = lowestCommonAncestor(root.right, p, q)
    if (root == p || root == q) return root
    else if (left && right) return root
    else if (left) return left
    else return right
}

# 二叉搜索树序列


function BSTSequences(root) {
    if (!root) return [[]]
    let res = []
    function dfs(cur, q, path) {
        if (cur.left) q.push(cur.left)
        if (cur.right) q.push(cur.right)
        if (!q.length) res.push(path)
        else {
            for (let i = 0; i < q.length; i++) {
                let newq = [...q.slice(0, i), ...q.slice(i + 1)]
                dfs(q[i], newq, [...path, [q[i].val]])
            }
        }
    }
    dfs(root, [], [root.val])
    return res
}

# 检查子树


function checkSubTree(t1, t2) {
    let arr1 = [], arr2 = []
    function dfs(t, arr) {
        if (!t) return
        dfs(t.left, arr)
        arr.push(t.val)
        dfs(t.right, arr)
    }
    dfs(t1, arr1)
    dfs(t2, arr2)
    let s1 = arr1.join(""), s2 = arr2.join("")
    if (s1.includes(s2)) return true
    return false
}

function checkSubTree(t1, t2) {
    if (!t1 && !t2) return true
    if (!t1 || !t2) return false
    function dfs(t1, t2) {
        if (!t1 && !t2) return true
        if (!t1 || !t2) return false
        return t1.val == t2.val && dfs(t1.left, t2.left) && dfs(t1.right, t2.right)
    }
    return dfs(t1, t2) || checkSubTree(t1.left, t2) || checkSubTree(t1.right, t2)
}

# 求和路径


function pathSum(root, sum) {
    let map = {}
    map[0] = 1
    function dfs(root, target, curSum) {
        if (!root) return 0
        curSum += root.val
        let res = 0
        res += map[curSum - target] ? map[curSum - target] : 0
        if (!map[curSum]) map[curSum] = 0
        map[curSum]++
        res += dfs(root.left, target, curSum)
        res += dfs(root.right, target, curSum)
        map[curSum]--
        return res
    }
    return dfs(root, sum, 0)
}

function pathSum(root, sum) {
    function dfs(r, s) {
        if (r) {
            s = s.slice().map(x => x + r.val)
            s.push(r.val)
            return s.filter(x => x == sum).length + dfs(r.left, s) + dfs(r.right, s)
        }
        return 0
    }
    return dfs(root, [])
}

function insertBits(N, M, i, j) {
    let insert = (1 << j - i + 1) - 1 << i
    return (N & ~insert) | M << i
}

# 下一个数


function findClosedNumbers(num) {
    function countsOne(num) {
        if (num < 0) return countsOne(-num)
        return num.toString(2).split('').filter(e => e === '1').length
    }
    let total = countsOne(num)
    let res = []
    let cur = num + 1
    while (countsOne(cur) !== total) cur++
    res[0] = cur
    cur = num - 1
    while (countsOne(cur) !== total) cur--
    res[1] = cur
    return res
}

# 整数转换


function convertInteger(A, B) {
    let ans = A ^ B
    let num = 0
    while (ans != 0) {
        ans &= (ans - 1)
        num++
    }
    return num
}

# 配对交换


function exchangeBits(num) {
    let temp1 = num >> 1
    let temp2 = num << 1
    let evenKeeper = 0, oddKeeper = 0
    for (let i = 0; i < 16; i++) {
        oddKeeper = oddKeeper | 2
        oddKeeper <<= 2
        evenKeeper = evenKeeper | 1
        evenKeeper <<= 2
    }
    oddKeeper = oddKeeper | 2
    evenKeeper >>= 2
    let num2 = (temp1 & evenKeeper) | (temp2 & oddKeeper)
    return num2
}

function exchangeBits(num) {

    odd = odd << 1
    even = even >>> 1

}

function pathWithObstacles(maze) {
    if (!maze.length) return []
    let path = [], map = new Map()
    if (getPath(maze, maze.length - 1, maze[0].length - 1, path, map)) return path
    return []
}

function getPath(maze, row, col, path, map) {
    if (row < 0 || col < 0 || maze[row][col] == 1) return false
    let point = [row, col]
    if (map.has(point)) return false
    let isAtOrigin = (row == 0) && (col == 0)
    if (isAtOrigin || getPath(maze, row, col - 1, path, map) || getPath(maze, row - 1, col, path, map)) {
        path.push([row, col])
        return true
    }
    map.set(point, true)
    return false
}

# 无重复字符串的排列组合


function permutation(S) {

        if (path === '') return res.push(cur)
        for (let i = 0, len = path.length; i < len; i++) {
            cur += path[i]
            backTrack(cur, path.slice(0, i).concat(path.slice(i + 1)))

        }
    }

}

# 括号


function generateParenthesis(n) {
    const ans = []
    function back(str, left, right) {
        if (left === n && right === n) return ans.push(str)
        if (left < n) back(str + '(', left + 1, right)
        if (left > right && right < n) back(str + ')', left, right + 1)
    }
    back('', 0, 0)
    return ans;
}

# 颜色填充


function floodFill(image, sr, sc, newColor) {
    const initVal = image[sr][sc]
    function dfs(sr, sc) {
        if (sr < 0 || sr === image.length ||
            sc < 0 || sc === image[0].length ||
            image[sr][sc] !== initVal || image[sr][sc] === newColor
        ) return

        image[sr][sc] = newColor;
        dfs(sr - 1, sc)
        dfs(sr, sc + 1)
        dfs(sr + 1, sc)
        dfs(sr, sc - 1)
    }
    dfs(sr, sc)
    return image
}

# 堆箱子


function pileBox(box) {
    box.sort((a, b) => b[2] - a[2])
    let dp = new Array(box.length)
    dp[0] = box[0][2]
    let res = dp[0]
    for (let i = 1; i < box.length; i++) {
        let max = 0
        for (let j = 0; j < i; j++) {
            if (box[j][0] > box[i][0] && box[j][1] > box[i][1] && box[j][2] > box[i][2]) {
                max = Math.max(max, dp[j])
            }
        }
        dp[i] = max + box[i][2]
        res = Math.max(res, dp[i])
    }
    return res
}

# 布尔运算


function countEval(s, result, memo = new Map()) {
    if (!s.length) return 0
    if (s.length == 1) return stringToBool(s) == result ? 1 : 0
    if (memo.has(result + s)) return memo.get(result + s)
    let ways = 0
    for (let i = 1; i < s.length; i += 2) {
        let c = s[i]
        let left = s.substring(0, i)
        let right = s.substring(i + 1, s.length)
        let leftTrue = countEval(left, true, memo)
        let leftFlase = countEval(left, false, memo)
        let rightTrue = countEval(right, true, memo)
        let rightFalse = countEval(right, false, memo)
        let total = (leftTrue + leftFlase) * (rightTrue + rightFalse)
        let totalTrue = 0
        if (c == "^") totalTrue = leftTrue * rightFalse + leftFlase * rightTrue
        else if (c == "&") totalTrue = leftTrue * rightTrue
        else if (c == "|") totalTrue = leftTrue * rightTrue + leftFlase * rightTrue + leftTrue * rightFalse

        let subways = result ? totalTrue : total - totalTrue
        ways += subways
    }
    memo.set(result + s, ways)
    return ways
}

function stringToBool(c) {
    return c == "1" ? true : false
}

# 合并排序的数组


function merge(A, m, B, n) {
    let pa = m - 1,
        pb = n - 1,
        tail = m + n - 1
    while (pa >= 0 || pb >= 0) {
        if (pa === -1) {
            A[tail] = B[pb--]
        } else if (pb === -1) {
            A[tail] = A[pa--]
        } else if (A[pa] < B[pb]) {
            A[tail] = B[pb--]
        } else {
            A[tail] = A[pa--]
        }
        tail--
    }
}

function groupAnagrams(strs) {

        const temp = item.split('').sort().join('')

# 搜索旋转数组


function search(arr, target) {
    if (!arr || arr.length === 0) return -1
    let left = 0, right = arr.length - 1
    while (left < right) {
        let mid = left + Math.floor((right - left) / 2)
        if (arr[left] < arr[mid]) {
            if (arr[left] <= target && target <= arr[mid]) right = mid;
            else left = mid
        } else if (arr[left] > arr[mid]) {
            if (arr[mid] <= target && target <= arr[right] && arr[left] > arr[right]) left = mid;
            else right = mid;
        } else {
            if (arr[left] !== target) left++;
            else right = left;
        }
    }
    return (arr[left] === target) ? left : -1
}

# 数字流的秩


class StreamRank {
    constructor() {
        this.arr = []
    }
    track(x) {
        let left = 0
        let right = this.arr.length - 1
        while (left <= right) {
            const mid = (left + right) >>> 1
            if (this.arr[mid] > x) {
                right = mid - 1
            } else if (this.arr[mid] < x) {
                left = mid + 1
            } else {
                left = mid
                break
            }
        }
        this.arr.splice(left, 0, x)
    }
    getRankOfNumber(x) {
        let left = 0
        let right = this.arr.length - 1
        while (left <= right) {
            const mid = (left + right) >>> 1
            if (this.arr[mid] > x) {
                right = mid - 1
            } else {
                left = mid + 1
            }
        }
        if (this.arr[left] !== x) left--
        return left + 1
    }
}

function wiggleSort(nums) {
    let flag = 0
    for (let i = 1; i < nums.length; i++) {
        if (flag == 0 && nums[i] > nums[i - 1]) {
            [nums[i], nums[i - 1]] = [nums[i - 1], nums[i]]
        }
        if (flag == 1 && nums[i] < nums[i - 1]) {
            [nums[i], nums[i - 1]] = [nums[i - 1], nums[i]]
        }
        flag = flag == 0 ? 1 : 0
    }
}

    } else {
        let k2 = (end2[1] - start2[1]) / (end2[0] - start2[0])
        let k1 = (end1[1] - start1[1]) / (end1[0] - start1[0])
        let b2 = (start2[1] - k2 * start2[0])
        let b1 = (start1[1] - k1 * start1[0])
        x = (b2 - b1) / (k1 - k2)
        y = (x * k1) + b1
    }
    let l = Math.max(Math.min(start1[0], end1[0]), Math.min(start2[0], end2[0]))
    let r = Math.min(Math.max(start1[0], end1[0]), Math.max(start2[0], end2[0]))
    let b = Math.max(Math.min(start1[1], end1[1]), Math.min(start2[1], end2[1]))
    let t = Math.min(Math.max(start1[1], end1[1]), Math.max(start2[1], end2[1]))
    if (x !== null && x >= l && x <= r && y <= t && y >= b) {
        return [x, y]
    } else {
        return []
    }
}

# 井字游戏


function tictactoe(board) {
    let size = 0, len = board.length, leftX = 0, rightX = 0, leftO = 0, rightO = 0
    let rowX = Array(len).fill(0), colX = Array(len).fill(0), rowO = Array(len).fill(0), colO = Array(len).fill(0)
    for (let i = 0; i < len; i++) {
        for (let j = 0; j < len; j++) {
            if (board[i][j] == 'X') {
                rowX[i]++; colX[j]++
                if (i == j) leftX++
                if (i + j == len - 1) rightX++
                size++
            } else if (board[i][j] == 'O') {
                rowO[i]++; colO[j]++
                if (i == j) leftO++
                if (i + j == len - 1) rightO++
                size++
            }
        }
    }
    if (leftX == len || rightX == len) return "X"
    if (leftO == len || rightO == len) return "O"
    for (let i = 0; i < len; i++) {
        if (rowX[i] == len || colX[i] == len) return "X"
        if (rowO[i] == len || colO[i] == len) return "O"
    }
    return size == len * len ? "Draw" : "Pending"
}

# 最小差


function smallestDifference(a, b) {
    let min = Infinity
    b.sort((a, b) => a - b)
    for (let i = 0; i < a.length; i++) {
        let start = 0,
            end = b.length
        mid = (start + end) / 2 | 0
        while (mid < end) {
            if (b[mid] == a[i]) break
            if (b[mid] > a[i]) {
                end = mid - 1
            } else {
                start = mid + 1
            }
            mid = (start + end) / 2 | 0
        }
        min = min > Math.abs(b[mid - 1] - a[i]) ? Math.abs(b[mid - 1] - a[i]) : min
        min = min > Math.abs(b[mid] - a[i]) ? Math.abs(b[mid] - a[i]) : min
        min = min > Math.abs(b[mid + 1] - a[i]) ? Math.abs(b[mid + 1] - a[i]) : min
    }
    return min
}

# 运算


class Operations {
    minus(a, b) {
        return a + this.negate(b)
    }
    multiply(a, b) {
        if (a < b) return this.multiply(b, a)
        let sum = 0
        for (let i = Math.abs(b); i > 0; i = this.minus(i, 1)) sum += a
        return b < 0 ? this.negate(sum) : sum
    }
    divide(a, b) {
        let absa = Math.abs(a)
        let absb = Math.abs(b)
        let product = 0, x = 0
        while (product + absb <= absa) {
            product += absb
            x++
        }
        if ((a < 0 && b < 0) || (a > 0 && b > 0)) return x
        else return this.negate(x)
    }
    negate(a) {
        let neg = 0
        let newSign = a < 0 ? 1 : -1
        let delta = newSign
        while (a != 0) {
            let differentSigns = (a + delta > 0) != (a > 0)
            if (a + delta != 0 && differentSigns) delta = newSign
            neg += delta
            a += delta
            delta += delta
        }
        return neg
    }
}

function maxAliveYear(birth, death) {
    const changes = new Array(102).fill(0)
    for (let i = 0, len = birth.length; i < len; i++) {

        changes[death[i] - 1900 + 1]--
    }

    for (let i = 0, len = changes.length; i < len; i++) {

# 平分正方形


function cutSquares(square1, square2) {
    let eps = 0.000000001;
    let x1m = square1[0] + square1[2] / 2.0;
    let y1m = square1[1] + square1[2] / 2.0;
    let x2m = square2[0] + square2[2] / 2.0;
    let y2m = square2[1] + square2[2] / 2.0;
    let x_max = Math.max(square1[0] + square1[2], square2[0] + square2[2]);
    let x_min = Math.min(square1[0], square2[0]);
    let y_max = Math.max(square1[1] + square1[2], square2[1] + square2[2]);
    let y_min = Math.min(square1[1], square2[1]);
    if (Math.abs(x2m - x1m) < eps) return [x1m, y_min, x1m, y_max]
    if (Math.abs(y2m - y1m) < eps) return [x_min, y1m, x_max, y1m]
    let k = (y2m - y1m) / (x2m - x1m);
    if (k > 1) return [x2m - (y2m - y_min) / k, y_min, x2m - (y2m - y_max) / k, y_max]
    else if (k < -1) return [x2m - (y2m - y_max) / k, y_max, x2m - (y2m - y_min) / k, y_min]
    else return [x_min, y2m - k * (x2m - x_min), x_max, y2m - k * (x2m - x_max)]
}

# 珠玑妙算


function masterMind(solution, guess) {
    let pure = 0, fake = 0, temp = new Map();
    solution.split('').forEach((e, i) => {
        if (e === guess[i]) pure += 1
        else temp.has(e) ? temp.set(e, temp.get(e) + 1) : temp.set(e, 1)
    })
    guess.split('').forEach((e, i) => {
        if (temp.has(e) && temp.get(e) > 0 && e !== solution[i]) {
            fake += 1
            temp.set(e, temp.get(e) - 1)
        }
    })
    return [pure, fake]
}

# 部分排序


function subSort(arr) {
    if (arr == null || arr.length === 0) return [-1, -1]
    let n = arr.length, max = arr[0], right = -1
    for (let i = 1; i < n; i++) {
        if (arr[i] >= max) max = arr[i]
        else right = i
    }
    if (right === -1) return [-1, -1]
    let min = arr[n - 1],
        left = -1
    for (let i = n - 2; i >= 0; i--) {
        if (arr[i] <= min) min = arr[i]
        else left = i
    }
    return [left, right]
}

function maxSubArray(nums) {

# 水域大小


function pondSizes(land) {
    let res
    let area
    res = []
    for (let i = 0; i < land.length; i++) {
        for (let j = 0; j < land[0].length; j++) {
            if (land[i][j] == 0) {
                area = 0
                dfs(land, i, j)
                res.push(area)
            }
        }
    }
    function dfs(land, i, j) {
        if (i < 0 || j < 0 || i >= land.length || j >= land[0].length) return
        if (land[i][j] == 0) {
            land[i][j] = 1
            area++
            dfs(land, i - 1, j)
            dfs(land, i + 1, j)
            dfs(land, i, j - 1)
            dfs(land, i, j + 1)
            dfs(land, i - 1, j - 1)
            dfs(land, i - 1, j + 1)
            dfs(land, i + 1, j - 1)
            dfs(land, i + 1, j + 1)
        }
    }
    return res.sort((a, b) => (a - b))
}

# 交换和


function findSwapValues(array1, array2) {
    let total1 = array1.reduce((total, current) => total += current, 0)
    let total2 = array2.reduce((total, current) => total += current, 0)
    let avg = (total1 + total2) / 2
    if (avg !== parseInt(avg)) return []
    array1 = [...new Set(array1)]
    array2 = [...new Set(array2)]
    for (let i = 0, len1 = array1.length; i < len1; i++) {
        for (let j = 0, len2 = array2.length; j < len2; j++) {
            if ((total1 - array1[i] + array2[j]) === (total2 + array1[i] - array2[j])) {
                return [array1[i], array2[j]]
            }
        }
    }
    return []
}

function pairSums(nums, target) {
    const res = []
    const map = new Map()
    for (let item of nums) {
        const num = map.get(target - item)
        if (num && num > 0) {

    }
    return res
}

# LRU缓存


class LRUCache {
    constructor(capacity) {
        this.capacity = capacity
        this.cache = new Map()
        this.keys = new Set()
    }
    get(key) {
        if (this.keys.has(key)) {
            this.keys.delete(key)
            this.keys.add(key)
            return this.cache.get(key)
        }
        else return -1
    }
    put(key, value) {
        if (this.keys.has(key)) {
            this.keys.delete(key)
        } else {
            if (this.keys.size === this.capacity) {
                let firstKey = this.keys.keys().next().value
                this.keys.delete(firstKey)
                this.cache.delete(firstKey)
            }
        }
        this.cache.set(key, value)
        this.keys.add(key)
    }
}

# 计算器


function calculate(s) {
    let str = s.replace(/ /g, '')
    let numArr = []
    let numStr = ''
    let symbol = ''
    let symbolArr = ["+", "-", "*", "/"]
    for (let i = 0; i <= str.length; i++) {
        const element = str[i]
        if (symbolArr.includes(element) || i === str.length) {
            switch (symbol) {
                case '*':
                    numArr.push(numArr.pop() * parseInt(numStr))
                    break
                case '/':
                    numArr.push(~~(numArr.pop() / parseInt(numStr)))
                    break
                case '-':
                    numArr.push(-parseInt(numStr))
                    break
                default:
                    numArr.push(parseInt(numStr))
                    break
            }
            symbol = element
            numStr = ''
            continue
        }
        numStr += element
    }
    let res = 0
    numArr.forEach(num => {
        res += num;
    })
    return res
}
